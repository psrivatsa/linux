" No Wrapping
: set nowrap
 
" Show line number
: set nu
 
" Set the syntax on
: syntax on
 
" Search options ; highlight and incremental search
: set hlsearch
: set incsearch
 
" Auto change directory
: set autochdir
 
" Set the ctags search from current to parent
: set tags=tags;
 
" Switching between windows using ALT+Arrow keys
: nmap <silent> <A-Up> :wincmd k<CR>
: nmap <silent> <A-Down> :wincmd j<CR>
: nmap <silent> <A-Left> :wincmd h<CR>
: nmap <silent> <A-Right> :wincmd l<CR>
 
" Set C style indentation
: set cindent

" System Verilog Syntax
: au BufNewFile,BufRead *.sv,*.svh,*.vh,*.v so ~/.vim/syntax/verilog_systemverilog.vim
